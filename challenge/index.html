Verify it yourself

from eth_account import Account
from eth_account.messages import encode_defunct
d = requests.get("https://rhinogent.com/fable_challenge_response.json").json()
s = d["onyx_signature"]
rec = Account.recover_message(encode_defunct(text=s["signed_digest"]), signature=s["signature"])
assert rec.lower() == s["signer_address"].lower()   # signed by the citizen it claims
